  // TODO: Narrate what goes on.
  document.addEventListener( "DOMContentLoaded", async() => {
    // Constants
    // 1. URL source of the JSON file
    const endpoint = 'https://gist.githubusercontent.com/Miserlou/c5cd8364bf9b2420bb29/raw/2bf258763cdddd704f8ffd3ea9a3e81d25e2c6f6/cities.json';
    // 2. Response, this will return a promise.
    const response = await fetch(endpoint);
    // 3. List of suggestions from HTML
    const suggestionsList = document.querySelector('.suggestions');

    // Variables
    let cities = [];
    let query = document.querySelector('.search').value;

    cities = await response.json(); // Wait for JSON response to request to GitHub server for 'cities' JSON file.
    
    // Functions
    function clear() { while(suggestionsList.firstChild) { suggestionsList.removeChild(suggestionsList.lastChild); } }
    
    function search() {
      query = document.querySelector('.search').value;
      let regex = new RegExp(query, 'gi');
      
      clear();
      if (query != "") { // DEBUG: console.log(query);
        for (city of cities){ // DEBUG: console.log(city.city);
          cityWithoutWhitespace = city.city.replace(/\s+/g, '');
          stateWithoutWhitespace = city.state.replace(/\s+/g, '');
          if (regex.test(city.city) || regex.test(cityWithoutWhitespace) || regex.test(city.state) || regex.test(stateWithoutWhitespace)) { 
            const cityListItem = document.createElement("li");
            cityListItem.innerHTML = `${city.city}, ${city.state}`;
            suggestionsList.appendChild(cityListItem);}
          }
        }
      else { // DEBUG: console.log(query);
        clear();
      }
    }

    // Main program
    clear();
    document.addEventListener("input",search);
  });
